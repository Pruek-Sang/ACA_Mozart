## **1.1 AutoLISP คืออะไร?**

**AutoLISP** = ภาษาโปรแกรมที่ฝังอยู่ใน AutoCAD ใช้สำหรับ:tiktok+3​

1. **สร้างวัตถุ (Entities):** เส้น, วงกลม, ข้อความ, Block
    
2. **อัตโนมัติ:** ทำงานซ้ำ ๆ ได้รวดเร็ว
    
3. **คำนวณ:** ทำคณิตศาสตร์, จัดการข้อมูล
    

**ตัวอย่างโค้ด AutoLISP:**

lisp

`; วาดเส้นตรง (command "LINE" '(0 0 0) '(100 100 0) "") ; วาดวงกลม (command "CIRCLE" '(50 50 0) 10) ; เขียนข้อความ (command "TEXT" '(10 10 0) 2.5 0 "Hello AutoCAD")`

## 💻 ส่วนที่ 2: Code Python — autolisp_generator.py

python

"""
autolisp_generator.py
======================
โมดูลสร้างโค้ด AutoLISP จากข้อมูล layout_coordinates.json

Features:
- อ่าน layout_coordinates.json จาก Module 7
- สร้างโค้ด AutoLISP สำหรับวาด:
  • Distribution Board (DB)
  • Outlets/Switches
  • Wire Routes (Conduit) แบบ Polyline
  • Labels (ข้อความบอกรายละเอียด)
- แยก Layer (DB, Wire, Outlet, Text)
- Export เป็น .lsp ไฟล์

Author: Volta (The Electrical Simulation Engineer Maid)
Version: 2.0.0
Date: 2025-11-17
"""

import json
from typing import Dict, List, Tuple, Optional
from pathlib import Path
from dataclasses import dataclass


# ======================== Configuration ========================

# AutoCAD Layers
LAYERS = {
    "DB": {"name": "E-DB", "color": 1, "linetype": "CONTINUOUS"},  # Red
    "WIRE": {"name": "E-WIRE", "color": 5, "linetype": "CONTINUOUS"},  # Blue
    "OUTLET": {"name": "E-OUTLET", "color": 3, "linetype": "CONTINUOUS"},  # Green
    "TEXT": {"name": "E-TEXT", "color": 7, "linetype": "CONTINUOUS"},  # White
    "DIMENSION": {"name": "E-DIM", "color": 8, "linetype": "CONTINUOUS"}  # Gray
}

# Symbols Scale
SYMBOL_SCALE = 0.5  # เมตร (ขนาดสัญลักษณ์)
TEXT_HEIGHT = 0.25  # เมตร (ความสูงตัวอักษร)


# ======================== AutoLISP Code Generator ========================

class AutoLISPGenerator:
    """
    คลาสสร้างโค้ด AutoLISP
    """
    
    def __init__(self, layout_json_path: str, output_dir: str = "./output"):
        """
        Initialize AutoLISP Generator
        
        Parameters:
        - layout_json_path: path to layout_coordinates.json
        - output_dir: output directory
        """
        self.layout_json_path = Path(layout_json_path)
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(parents=True, exist_ok=True)
        
        # Load layout data
        with open(self.layout_json_path, 'r', encoding='utf-8') as f:
            self.layout_data = json.load(f)
        
        # AutoLISP Code Storage
        self.lisp_code: List[str] = []
        
        print(f"✅ Loaded layout data from: {self.layout_json_path}")
        print(f"   Project: {self.layout_data.get('project_name', 'Unknown')}")
        print(f"   Circuits: {len(self.layout_data.get('circuits', []))}")
    
    def _add_header(self):
        """เพิ่ม Header ของโค้ด AutoLISP"""
        project_name = self.layout_data.get('project_name', 'Electrical Layout')
        
        header = f"""
;;;=========================================================================
;;; AUTO-GENERATED AUTOLISP CODE
;;; Project: {project_name}
;;; Generated by: MCP Core Module 8 (AutoLISP Generator)
;;; Author: Volta
;;; Date: {self._get_current_datetime()}
;;;=========================================================================

(defun C:DRAW-ELECTRICAL-LAYOUT ()
  (princ "\\n>>> Starting Electrical Layout Drawing...")
  
  ; Set command echo off
  (setvar "CMDECHO" 0)
  
  ; Create Layers
  (create-electrical-layers)
  
  ; Draw Distribution Boards
  (draw-distribution-boards)
  
  ; Draw Wire Routes
  (draw-wire-routes)
  
  ; Draw Outlets
  (draw-outlets)
  
  ; Add Labels
  (add-labels)
  
  (princ "\\n>>> Electrical Layout Drawing Completed!")
  (princ)
)

;;;=========================================================================
;;; LAYER CREATION
;;;=========================================================================

(defun create-electrical-layers ()
  (princ "\\n  Creating layers...")
"""
        
        # Add layer creation code
        for layer_key, layer_info in LAYERS.items():
            header += f"""
  (if (not (tblsearch "LAYER" "{layer_info['name']}"))
    (command "LAYER" "N" "{layer_info['name']}" "C" "{layer_info['color']}" "{layer_info['name']}" "LT" "{layer_info['linetype']}" "{layer_info['name']}" "")
  )"""
        
        header += """
  (princ "\\n  Layers created successfully!")
)

"""
        
        self.lisp_code.append(header)
    
    def _add_db_drawing_function(self):
        """สร้างฟังก์ชันวาด Distribution Board"""
        
        func = """
;;;=========================================================================
;;; DISTRIBUTION BOARD DRAWING
;;;=========================================================================

(defun draw-distribution-boards ()
  (princ "\\n  Drawing Distribution Boards...")
  (setvar "CLAYER" "E-DB")
  
"""
        
        # หา DB positions จาก circuits
        db_positions = set()
        for circuit in self.layout_data.get('circuits', []):
            db_pos = circuit.get('db_position')
            if db_pos:
                db_positions.add((db_pos['x'], db_pos['y'], db_pos['z']))
        
        # วาด DB แต่ละตัว
        for i, (x, y, z) in enumerate(db_positions, 1):
            func += f"""
  ; DB{i} at ({x}, {y}, {z})
  (command "RECTANGLE" 
    (list {x - SYMBOL_SCALE/2} {y - SYMBOL_SCALE/2} {z})
    (list {x + SYMBOL_SCALE/2} {y + SYMBOL_SCALE/2} {z})
  )
  
  ; DB Label
  (command "TEXT" "J" "MC" 
    (list {x} {y} {z})
    {TEXT_HEIGHT}
    0
    "DB{i}"
  )
"""
        
        func += """
  (princ "\\n  Distribution Boards drawn!")
)

"""
        
        self.lisp_code.append(func)
    
    def _add_wire_routes_function(self):
        """สร้างฟังก์ชันวาดเส้นสาย"""
        
        func = """
;;;=========================================================================
;;; WIRE ROUTES DRAWING
;;;=========================================================================

(defun draw-wire-routes ()
  (princ "\\n  Drawing wire routes...")
  (setvar "CLAYER" "E-WIRE")
  
"""
        
        for circuit in self.layout_data.get('circuits', []):
            circuit_id = circuit.get('circuit_id', 'Unknown')
            circuit_name = circuit.get('circuit_name', 'Unknown')
            waypoints = circuit.get('waypoints', [])
            wire_size = circuit.get('wire_size_mm2', 0)
            conduit_size = circuit.get('conduit_size_inch', '')
            
            if len(waypoints) < 2:
                continue
            
            func += f"""
  ; Circuit: {circuit_id} - {circuit_name}
  ; Wire: THW {wire_size} mm², Conduit: {conduit_size}"
  (command "PLINE"
"""
            
            # Add waypoints
            for wp in waypoints:
                func += f"""    (list {wp['x']} {wp['y']} {wp['z']})
"""
            
            func += f"""    ""
  )
  
  ; Label at midpoint
  (setvar "CLAYER" "E-TEXT")
  (command "TEXT"
    (list {waypoints[len(waypoints)//2]['x']} {waypoints[len(waypoints)//2]['y']} {waypoints[len(waypoints)//2]['z']})
    {TEXT_HEIGHT * 0.8}
    0
    "{circuit_id}: {wire_size}mm²"
  )
  (setvar "CLAYER" "E-WIRE")
  
"""
        
        func += """
  (princ "\\n  Wire routes drawn!")
)

"""
        
        self.lisp_code.append(func)
    
    def _add_outlets_function(self):
        """สร้างฟังก์ชันวาด Outlets"""
        
        func = """
;;;=========================================================================
;;; OUTLETS DRAWING
;;;=========================================================================

(defun draw-outlets ()
  (princ "\\n  Drawing outlets...")
  (setvar "CLAYER" "E-OUTLET")
  
"""
        
        for circuit in self.layout_data.get('circuits', []):
            circuit_id = circuit.get('circuit_id', 'Unknown')
            outlet_pos = circuit.get('outlet_position')
            
            if not outlet_pos:
                continue
            
            x, y, z = outlet_pos['x'], outlet_pos['y'], outlet_pos['z']
            
            func += f"""
  ; Outlet for {circuit_id}
  (command "CIRCLE"
    (list {x} {y} {z})
    {SYMBOL_SCALE * 0.3}
  )
  
  ; Outlet Label
  (setvar "CLAYER" "E-TEXT")
  (command "TEXT" "J" "MC"
    (list {x} {y + SYMBOL_SCALE * 0.5} {z})
    {TEXT_HEIGHT * 0.7}
    0
    "{circuit_id}"
  )
  (setvar "CLAYER" "E-OUTLET")
  
"""
        
        func += """
  (princ "\\n  Outlets drawn!")
)

"""
        
        self.lisp_code.append(func)
    
    def _add_labels_function(self):
        """สร้างฟังก์ชันเพิ่ม Labels"""
        
        func = """
;;;=========================================================================
;;; LABELS
;;;=========================================================================

(defun add-labels ()
  (princ "\\n  Adding labels...")
  (setvar "CLAYER" "E-TEXT")
  
  ; Project Title
  (command "TEXT" "J" "TL"
    (list 0 0 0)
"""
        func += f"""    {TEXT_HEIGHT * 1.5}
    0
    "{self.layout_data.get('project_name', 'Electrical Layout')}"
  )
  
  (princ "\\n  Labels added!")
)

"""
        
        self.lisp_code.append(func)
    
    def _add_footer(self):
        """เพิ่ม Footer"""
        
        footer = """
;;;=========================================================================
;;; END OF FILE
;;;=========================================================================

; Run the main function automatically
(princ "\\n>>> Type 'DRAW-ELECTRICAL-LAYOUT' to start drawing.")
(princ)

; Auto-run (optional - uncomment to run automatically)
; (C:DRAW-ELECTRICAL-LAYOUT)
"""
        
        self.lisp_code.append(footer)
    
    def generate(self) -> str:
        """สร้างโค้ด AutoLISP ทั้งหมด"""
        
        print("\n📝 Generating AutoLISP code...")
        
        self.lisp_code = []
        
        # Add components
        self._add_header()
        self._add_db_drawing_function()
        self._add_wire_routes_function()
        self._add_outlets_function()
        self._add_labels_function()
        self._add_footer()
        
        # Combine all code
        full_code = "\n".join(self.lisp_code)
        
        print("✅ AutoLISP code generated!")
        
        return full_code
    
    def save_lisp_file(self, filename: str = "electrical_layout.lsp") -> Path:
        """บันทึกเป็นไฟล์ .lsp"""
        
        filepath = self.output_dir / filename
        
        code = self.generate()
        
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(code)
        
        print(f"✅ Saved AutoLISP file: {filepath}")
        
        return filepath
    
    def _get_current_datetime(self) -> str:
        """ดึงวันที่/เวลาปัจจุบัน"""
        from datetime import datetime
        return datetime.now().strftime("%Y-%m-%d %H:%M:%S")


# ======================== Advanced Features ========================

class AutoLISPAdvancedGenerator(AutoLISPGenerator):
    """
    เวอร์ชันขยาย — มีฟีเจอร์เพิ่มเติม
    """
    
    def add_single_line_diagram(self):
        """
        สร้าง Single Line Diagram (SLD)
        
        แสดงวงจรแบบง่าย:
        - DB (สี่เหลี่ยม)
        - Breaker (เส้น)
        - Load (สัญลักษณ์)
        """
        
        sld_func = """
;;;=========================================================================
;;; SINGLE LINE DIAGRAM
;;;=========================================================================

(defun draw-single-line-diagram ()
  (princ "\\n  Drawing Single Line Diagram...")
  
  ; Starting position
  (setq sld-start-x 100)
  (setq sld-start-y 50)
  (setq sld-spacing 5)
  
  ; Draw DB
  (command "RECTANGLE"
    (list sld-start-x sld-start-y 0)
    (list (+ sld-start-x 3) (+ sld-start-y 8) 0)
  )
  (command "TEXT" "J" "MC"
    (list (+ sld-start-x 1.5) (+ sld-start-y 4) 0)
    0.5
    0
    "MAIN DB"
  )
  
"""
        
        # Draw circuits in SLD
        y_offset = 0
        for i, circuit in enumerate(self.layout_data.get('circuits', []), 1):
            circuit_name = circuit.get('circuit_name', 'Unknown')
            breaker_rating = circuit.get('breaker_rating_a', 0)
            
            sld_func += f"""
  ; Circuit {i}: {circuit_name}
  (command "LINE"
    (list (+ sld-start-x 3) (+ sld-start-y {4 + y_offset}) 0)
    (list (+ sld-start-x 6) (+ sld-start-y {4 + y_offset}) 0)
    ""
  )
  
  ; Breaker symbol
  (command "CIRCLE"
    (list (+ sld-start-x 4.5) (+ sld-start-y {4 + y_offset}) 0)
    0.3
  )
  (command "TEXT"
    (list (+ sld-start-x 4.5) (+ sld-start-y {4 + y_offset + 0.5}) 0)
    0.2
    0
    "{breaker_rating}A"
  )
  
  ; Load label
  (command "TEXT"
    (list (+ sld-start-x 6.5) (+ sld-start-y {4 + y_offset}) 0)
    0.25
    0
    "{circuit_name}"
  )
"""
            
            y_offset -= 1.5
        
        sld_func += """
  (princ "\\n  Single Line Diagram drawn!")
)

"""
        
        self.lisp_code.insert(-1, sld_func)  # Insert before footer
        
        # Add to main function call
        main_func_index = next((i for i, code in enumerate(self.lisp_code) if "draw-outlets)" in code), None)
        if main_func_index:
            insert_line = "\n  ; Draw Single Line Diagram\n  (draw-single-line-diagram)\n"
            self.lisp_code[main_func_index] += insert_line
    
    def add_legend(self):
        """เพิ่ม Legend (คำอธิบายสัญลักษณ์)"""
        
        legend_func = """
;;;=========================================================================
;;; LEGEND
;;;=========================================================================

(defun draw-legend ()
  (princ "\\n  Drawing legend...")
  
  (setq legend-x 0)
  (setq legend-y -10)
  
  ; Title
  (command "TEXT" "J" "TL"
    (list legend-x legend-y 0)
    0.4
    0
    "LEGEND"
  )
  
  ; DB Symbol
  (command "RECTANGLE"
    (list legend-x (- legend-y 2) 0)
    (list (+ legend-x 0.5) (- legend-y 2.5) 0)
  )
  (command "TEXT"
    (list (+ legend-x 1) (- legend-y 2.25) 0)
    0.25
    0
    "Distribution Board (DB)"
  )
  
  ; Outlet Symbol
  (command "CIRCLE"
    (list (+ legend-x 0.25) (- legend-y 3.5) 0)
    0.15
  )
  (command "TEXT"
    (list (+ legend-x 1) (- legend-y 3.5) 0)
    0.25
    0
    "Outlet / Switch"
  )
  
  ; Wire Symbol
  (command "LINE"
    (list legend-x (- legend-y 4.5) 0)
    (list (+ legend-x 0.5) (- legend-y 4.5) 0)
    ""
  )
  (command "TEXT"
    (list (+ legend-x 1) (- legend-y 4.5) 0)
    0.25
    0
    "Wire Route (Conduit)"
  )
  
  (princ "\\n  Legend drawn!")
)

"""
        
        self.lisp_code.insert(-1, legend_func)

## 📊 ส่วนที่ 3: ตัวอย่างโค้ด AutoLISP ที่สร้างได้

lisp

`;;;;=========================================================================
;;; AUTO-GENERATED AUTOLISP CODE
;;; Project: บ้าน 2 ชั้น Modern Contemporary
;;; Generated by: MCP Core Module 8 (AutoLISP Generator)
;;; Author: Volta
;;; Date: 2025-11-17 01:00:00
;;;=========================================================================

(defun C:DRAW-ELECTRICAL-LAYOUT ()
  (princ "\n>>> Starting Electrical Layout Drawing...")
  
  ; Set command echo off
  (setvar "CMDECHO" 0)
  
  ; Create Layers
  (create-electrical-layers)
  
  ; Draw Distribution Boards
  (draw-distribution-boards)
  
  ; Draw Wire Routes
  (draw-wire-routes)
  
  ; Draw Outlets
  (draw-outlets)
  
  ; Add Labels
  (add-labels)
  
  (princ "\n>>> Electrical Layout Drawing Completed!")
  (princ)
)

;;;=========================================================================
;;; LAYER CREATION
;;;=========================================================================

(defun create-electrical-layers ()
  (princ "\n  Creating layers...")

  (if (not (tblsearch "LAYER" "E-DB"))
    (command "LAYER" "N" "E-DB" "C" "1" "E-DB" "LT" "CONTINUOUS" "E-DB" "")
  )

  (if (not (tblsearch "LAYER" "E-WIRE"))
    (command "LAYER" "N" "E-WIRE" "C" "5" "E-WIRE" "LT" "CONTINUOUS" "E-WIRE" "")
  )

  (if (not (tblsearch "LAYER" "E-OUTLET"))
    (command "LAYER" "N" "E-OUTLET" "C" "3" "E-OUTLET" "LT" "CONTINUOUS" "E-OUTLET" "")
  )

  (if (not (tblsearch "LAYER" "E-TEXT"))
    (command "LAYER" "N" "E-TEXT" "C" "7" "E-TEXT" "LT" "CONTINUOUS" "E-TEXT" "")
  )
  
  (princ "\n  Layers created successfully!")
)

;;;=========================================================================
;;; DISTRIBUTION BOARD DRAWING
;;;=========================================================================

(defun draw-distribution-boards ()
  (princ "\n  Drawing Distribution Boards...")
  (setvar "CLAYER" "E-DB")
  

  ; DB1 at (0, 6, 0)
  (command "RECTANGLE" 
    (list -0.25 5.75 0)
    (list 0.25 6.25 0)
  )
  
  ; DB Label
  (command "TEXT" "J" "MC" 
    (list 0 6 0)
    0.25
    0
    "DB1"
  )

  (princ "\n  Distribution Boards drawn!")
)

;;;=========================================================================
;;; WIRE ROUTES DRAWING
;;;=========================================================================

(defun draw-wire-routes ()
  (princ "\n  Drawing wire routes...")
  (setvar "CLAYER" "E-WIRE")
  

  ; Circuit: C01 - แอร์ ห้องนั่งเล่น
  ; Wire: THW 2.5 mm², Conduit: 1/2"
  (command "PLINE"
    (list 0 6 0)
    (list 6 6 0)
    (list 6 4 2.5)
    ""
  )
  
  ; Label at midpoint
  (setvar "CLAYER" "E-TEXT")
  (command "TEXT"
    (list 6 6 0)
    0.2
    0
    "C01: 2.5mm²"
  )
  (setvar "CLAYER" "E-WIRE")
  

  (princ "\n  Wire routes drawn!")
)

;;;=========================================================================
;;; OUTLETS DRAWING
;;;=========================================================================

(defun draw-outlets ()
  (princ "\n  Drawing outlets...")
  (setvar "CLAYER" "E-OUTLET")
  

  ; Outlet for C01
  (command "CIRCLE"
    (list 6 4 2.5)
    0.15
  )
  
  ; Outlet Label
  (setvar "CLAYER" "E-TEXT")
